The tests are failing because of a classic integration testing hurdle: **The Cart is empty**. Since your `checkout` service requires items in the cart to create an order, and we didn't add any products to User A's cart in the `beforeAll` block, the checkout failed, `orderAId` became undefined, and the subsequent GET requests hit `404`.

Here is how we fix the test and update Swagger.

### 🛠️ 1. Fix the Integration Test

We need to ensure User A has a product in their cart before calling checkout.

```ts
// tests/integration/orders/order-details.test.ts
import { describe, it, expect, beforeAll } from "vitest";
import request from "supertest";
import app from "../../../src/app.js";
import { getCustomerToken } from "../../helpers/auth.helper.js";
import { prisma } from "../../../src/config/prisma.js";

describe("Order Details Access Control", () => {
  let userAToken: string;
  let userBToken: string;
  let orderAId: string;

  beforeAll(async () => {
    userAToken = await getCustomerToken("userA@test.com", "password123");
    userBToken = await getCustomerToken("userB@test.com", "password123");

    // 1. Ensure a product exists in the DB
    const product = await prisma.product.create({
      data: {
        name: "Test Laptop",
        price: 999,
        stock: 10,
        description: "High end laptop",
        categoryId: (await prisma.category.findFirst())?.id || "",
      },
    });

    // 2. Add product to User A's cart
    await request(app)
      .post("/api/cart")
      .set("Authorization", `Bearer ${userAToken}`)
      .send({ productId: product.id, quantity: 1 });

    // 3. Now checkout works
    const res = await request(app)
      .post("/api/orders/checkout")
      .set("Authorization", `Bearer ${userAToken}`);

    orderAId = res.body.id;
  });

  it("should allow User A to view their own order", async () => {
    const res = await request(app)
      .get(`/api/orders/${orderAId}`)
      .set("Authorization", `Bearer ${userAToken}`);

    expect(res.statusCode).toBe(200);
    expect(res.body.id).toBe(orderAId);
    expect(res.body).toHaveProperty("items");
  });

  it("should prevent User B from viewing User A's order", async () => {
    const res = await request(app)
      .get(`/api/orders/${orderAId}`)
      .set("Authorization", `Bearer ${userBToken}`);

    expect(res.statusCode).toBe(403);
    expect(res.body.error.message).toMatch(/Forbidden|own/i);
  });
});
```

"/api/orders/{id}": {

    responses: {
      200: {
        description: "Order details retrieved successfully",
        content: {
          "application/json": {
            schema: {
              type: "object",
              properties: {
                id: { type: "string" },
                totalAmount: { type: "number" },
                status: { type: "string" },
                items: { type: "array", items: { type: "object" } }
              }
            }
          }
        }
      },
      403: { description: "Forbidden - Not your order" },
      404: { description: "Order not found" }
    }
  }
}
